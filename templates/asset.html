""" cur.execute(update_query, (status, ram, storage, location, updated_logs, now, id)) conn.commit() return redirect(url_for('assets')) placeholder = "%s" if DB_URL else "?" cur.execute(f"SELECT * FROM assets WHERE id = {placeholder}", (id,)) asset = cur.fetchone() cur.close() conn.close() return render_template('edit.html', asset=asset) @app.route('/delete/') def delete(id): placeholder = "%s" if DB_URL else "?" conn = get_db_connection() cur = conn.cursor() cur.execute(f"DELETE FROM assets WHERE id = {placeholder}", (id,)) conn.commit() cur.close() conn.close() return redirect(url_for('assets')) @app.route('/qr/') def show_qr(id): base_url = "https://asset-tracker-system-o5zl.onrender.com" url = f"{base_url}/asset/{id}" qr = qrcode.make(url) buffer = io.BytesIO() qr.save(buffer, format="PNG") img_str = base64.b64encode(buffer.getvalue()).decode() return render_template('qr_display.html', img_str=img_str, id=id) @app.route('/asset/') def asset(id): conn = get_db_connection() cur = conn.cursor() placeholder = "%s" if DB_URL else "?" cur.execute(f"SELECT * FROM assets WHERE id={placeholder}", (id,)) data = cur.fetchone() # Handle scan count update new_count = data[8] + 1 # scan_count index cur.execute(f"UPDATE assets SET scan_count={placeholder} WHERE id={placeholder}", (new_count, id)) conn.commit() cur.close() conn.close() return render_template('asset.html', data=data) if __name__ == '__main__': app.run()
